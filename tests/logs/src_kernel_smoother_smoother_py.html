8def kern_smooth(x, y, K="gaussian", h=None): 

9 """ 

10 Perform a kernel-based smoothing 

11 

12 parameters: 

13 ---------- 

14 x : np.ndarray 

15 the x values 

16 y : np.ndarray 

17 the y values to be smoothed 

18 K : str 

19 the name of the kernel method to be applied, 

20 one of the following methods: "gaussian", "triangular", "Epanechnikov" 

21 h : int or float 

22 the bandwidth 

23 

24 Raises: 

25 ------ 

26 Exception, in case of unsupported kernel name 

27 

28 Returns: 

29 ------- 

30 a np.ndarray of y-smoothed values 

31 """ 

32 

33 def _smooth(w, y): 

34 return 1/sum(w) * w.dot(y) 

35 

36 valid_kernels = ["gaussian", "triangular", "Epanechnikov"] 

37 if K not in valid_kernels: 

38 error_msg = "kernel must be one of the following: {}" \ 

39 .format(", ".join(valid_kernels)) 

40 raise Exception(error_msg) 

41 

42 if h is None: 

43 h = bandwidth(y) 

44 W = [eval(K)(x0, x, h) for x0 in x] 

45 return np.array([_smooth(w, y) for w in W])
